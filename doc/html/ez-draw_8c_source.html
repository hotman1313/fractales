524 /*
525  * Main loop. To break, just call ez_quit().
526  * This function displays the windows, then wait for events and dispatch them
527  * by calling the corresponding callbacks.
528  * Once returned from ez_main_loop(), no more graphic call should be done.
529 */
530 
531 void ez_main_loop (void)
532 {
533  int i;
534 #ifdef EZ_BASE_XLIB
535  Ez_event ev;
536 #elif defined EZ_BASE_WIN32
537  MSG msg;
538 #endif /* EZ_BASE_ */
539 
540  if (ez_state == EZ_PRE_INIT) {
541  ez_error ("ez_main_loop: error, ez_init must be called first\n");
542  return;
543  }
544  if (ez_state >= EZ_MAIN_LOOP) {
545  ez_error ("ez_main_loop: error, called several times\n");
546  return;
547  }
548  ez_state = EZ_MAIN_LOOP;
549 
550  /* Display all windows whose displaying was delayed */
551  for (i = 0; i < ezx.win_nb; i++) {
552  Ez_win_info *info;
553  if (ez_info_get (ezx.win_l[i], &info) < 0) continue;
554  if (info->show) ez_window_show (ezx.win_l[i], 1);
555  }
556 
557  /* Wait for next event then call the callback */
558  while (ezx.main_loop != 0 && ezx.win_nb > 0) {
559 #ifdef EZ_BASE_XLIB
560  ez_event_next (&ev);
561  ez_event_dispatch (&ev);
562 #elif defined EZ_BASE_WIN32
563  ez_msg_next (&msg);
564  DispatchMessage (&msg);
565 #endif /* EZ_BASE_ */
566  }
567 
568  ez_state = EZ_FINAL;
569 }

620 /*
621  * Convert a color from HSV to RGB ; h in [0..360], s,v in [0..1]
622  * see http://en.wikipedia.org/wiki/Hue/saturation/value
623 */
624 
625 void ez_HSV_to_RGB (double h, double s, double v,
626  Ez_uint8 *r, Ez_uint8 *g, Ez_uint8 *b)
627 {
628  int ti = ((int) h / 60) % 6;
629  double f = h / 60 - ti;
630  v *= 255;
631  switch (ti) {
632  case 0 : *r = v; *b = v*(1-s); *g = v*(1-(1-f)*s); break;
633  case 1 : *g = v; *b = v*(1-s); *r = v*(1- f *s); break;
634  case 2 : *g = v; *r = v*(1-s); *b = v*(1-(1-f)*s); break;
635  case 3 : *b = v; *r = v*(1-s); *g = v*(1 -f *s); break;
636  case 4 : *b = v; *g = v*(1-s); *r = v*(1-(1-f)*s); break;
637  case 5 : *r = v; *g = v*(1-s); *b = v*(1- f *s); break;
638  default : *r = *g = *b = 0;
639  }
640 }

914 /*
915  * Display text; same usage as printf.
916  *
917  * Example : ez_draw_text (win, EZ_TL, x, y,
918  * "Width = %d\nHeight = %d", w, h);
919  *
920  * The coordinates x1,y1 are relative to align, see Ez_Align definition.
921  * align also specifies if the background is filled or not.
922  *
923  * By default:
924  * - the text is displayed with the font number 0 ("6x13") ;
925  * can be changed with ez_set_nfont().
926  * - text is displayed in black; can be changed with ez_set_color().
927 */
928 
929 void ez_draw_text (Ez_window win, Ez_Align align, int x1, int y1,
930  const char *format, ...)
931 {
932  int valign, halign, fillbg;
933  va_list (ap);
934  char buf[16384];
935  int i, j, k, n, x, y, a, b, c;
936 
937 #ifdef EZ_BASE_XLIB
938  XFontStruct *font = ezx.font[ezx.nfont];
939 #elif defined EZ_BASE_WIN32
940  TEXTMETRIC text_metric;
941 #endif /* EZ_BASE_ */
942 
943  if (align <= EZ_AA || align == EZ_BB || align >= EZ_CC)
944  { ez_error ("ez_draw_text: bad align\n"); return; }
945 
946  /* Decode align */
947  fillbg = 0;
948  if (align > EZ_BB) { fillbg = 1; align -= 10; }
949  align -= EZ_AA + 1;
950  halign = align % 3;
951  valign = align / 3;
952 
953  /* Print the formated string in buf */
954  va_start (ap, format);
955  vsnprintf (buf, sizeof(buf)-1, format, ap);
956  va_end (ap);
957  buf[sizeof(buf)-1] = 0;
958  if (buf[0] == 0) return;
959 
960  /* Count the number of lines */
961  for (i = j = k = 0; ; i++)
962  if (buf[i] == '\n' || buf[i] == 0) {
963  k++; j = i+1;
964  if (buf[i] == 0) break;
965  }
966  n = k;
967 
968 #ifdef EZ_BASE_XLIB
969 
970  if (win == ezx.dbuf_win) win = ezx.dbuf_pix;
971  a = font->ascent; b = font->descent; c = a+b+b;
972 
973  /* Display line by line */
974  for (i = j = k = 0; ; i++)
975  if (buf[i] == '\n' || buf[i] == 0) {
976  x = x1 - XTextWidth (font, buf+j, i-j) * halign/2;
977  y = y1 + a + c*k - (c*n-b) * valign/2;
978  if (fillbg == 0)
979  XDrawString (ezx.display, win, ezx.gc, x, y, buf+j, i-j);
980  else XDrawImageString (ezx.display, win, ezx.gc, x, y, buf+j, i-j);
981  k++; j = i+1;
982  if (buf[i] == 0) break;
983  }
984 
985 #elif defined EZ_BASE_WIN32
986 
987  ez_cur_win (win);
988  GetTextMetrics (ezx.hdc, &text_metric);
989  a = text_metric.tmAscent; b = text_metric.tmDescent; c = a+b;
990 
991  if (fillbg == 0) SetBkMode (ezx.hdc, TRANSPARENT);
992 
993  /* Display line by line */
994  for (i = j = k = 0; ; i++)
995  if (buf[i] == '\n' || buf[i] == 0) {
996  x = x1 - (i-j)*text_metric.tmAveCharWidth * halign/2;
997  y = y1 + c*k - (c*n-b) * valign/2 -2;
998  TextOut (ezx.hdc, x, y, buf+j, i-j);
999  k++; j = i+1;
1000  if (buf[i] == 0) break;
1001  }
1002 
1003  /* Restore background drawing mode */
1004  if (fillbg == 0) SetBkMode (ezx.hdc, OPAQUE);
1005 
1006 #endif /* EZ_BASE_ */
1007 }

1231  * Insert a timer in the list. Return 0 on success, -1 on error.
1232 */
1233 
1234 int ez_timer_add (Ez_window win, int delay)
1235 {
1236  int m1, m2, mid;
1237  struct timeval t, *mt;
1238 
1239  ez_timer_remove (win);
1240 
1241  if (ezx.timer_nb >= EZ_TIMER_MAX) {
1242  ez_error ("ez_timer_add: too many timers\n");
1243  return -1;
1244  }
1245 
1246  /* Retrieve current date */
1247  ez_gettimeofday (&t);
1248 
1249  /* Compute expiration date */
1250  t.tv_usec += delay * 1000; /* delay in milliseconds */
1251  if (t.tv_usec > 1000000) {
1252  t.tv_sec += t.tv_usec / 1000000;
1253  t.tv_usec %= 1000000;
1254  }
1255 
1256  /* Dichotomic search */
1257  m1 = 0; m2 = ezx.timer_nb;
1258  while (m2-m1 > 0) {
1259  mid = (m1+m2) / 2; /* Middle s.t. m1 <= mid < m2 */
1260  mt = &ezx.timer_l[mid].expiration;
1261  if ( mt->tv_sec < t.tv_sec ||
